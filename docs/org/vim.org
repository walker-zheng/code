   - sdcv
#+BEGIN_SRC emacs-lisp
fu Sdcv() range
    let l:saved_reg = @"
    execute "normal! vgvy"
    let l:Command = substitute(@", "[\n$.,;]", "", "g")
    execute "!sdcv '" . l:Command . "'"
    let @" = l:saved_reg
endfu
map <leader>k :!sdcv <C-R>=substitute(expand("<cword>"), "[,.;\n$]", "", "g")<CR><CR>
vnoremap <leader>k :call Sdcv()<CR>
#+END_SRC
